PartialTracker FFT information into Array

by Federico Reuben


PartialTracker( windowSize,  numberPartials )


Uses Josh's PV_MagBuffer and PV_FreqBuffer to get FFT information of an audio input into an Array. You can specify the amount of partials you want from maximum to minimum.


an example:


s.boot;


p = PartialTracker(512, 5); 

p.startsynth(1); //use AudioIn input - channel 1

p.getarrays;

p.frequencies; //get array with frequencies

p.magnitudes; //get array with magnitudes

p.numpartials(80); //change number of partials

p.list


p.numpartials(12)

p.frequencies.cpsmidi.round(1).midinote //note names



r = Routine({

inf.do({

p.frequencies.cpsmidi.round(1).midinote.postln;

p.numpartials(rrand(1,12));

[0.2,0.4,0.1,0.5,0.75,1].choose.yield;

})

})


r.play;

r.stop;



(

MIDIClient.init;

MIDIClient.init(MIDIClient.sources.size, MIDIClient.destinations.size);

MIDIClient.sources.size.do({ |i|

MIDIIn.connect(i, MIDIClient.sources[i]);

});

)


m = MIDIOut(0, MIDIClient.destinations.at(0).uid);



(

c = Pbind(\midinote, Pseq.new((p.frequencies.cpsmidi.round(1)).postln, 1), \amp, Pseq.new(p.magnitudes.linlin(0,127,0,12), 1), \dur, Pseq.new(Array.fill(p.frequencies.size, 0.15), 1)); 

z = c <> (type: \midi, midiout:  m);  

1.do{z.next(Event.new).play}; 

)


p.numpartials(24);

p.stoptask // stop getting info



(

MIDIClient.init;

MIDIClient.init(MIDIClient.sources.size, MIDIClient.destinations.size);

MIDIClient.sources.size.do({ |i|

MIDIIn.connect(i, MIDIClient.sources[i]);

});

)


m = MIDIOut(0, MIDIClient.destinations.at(0).uid);


(

r = Routine({inf.do({

c = Pbind(\midinote, Pseq.new((p.frequencies.cpsmidi.round(1)).postln, 1), \amp, Pseq.new(p.magnitudes.linlin(0,127,0,400), 1), \dur, 0.0); 

z = c <> (type: \midi, midiout:  m);  

1.do{z.next(Event.new).play}; 

~time.yield;

});

})

)



~time = 0.12;

r.play;

r.stop;



p.numpartials(10);

p.startsynth2(7); //use AudioIn input channel 7

p.thresh(1)

p.getarrays;

p.frequencies; //get array with frequencies

p.magnitudes; //get array with magnitudes

p.numpartials(10); //change number of partials

p.list
